def performance_metric2(y_true, y_predict):
    """"""
    ss_tot = 0
    ss_res = 0
    _y = np.mean(y_true)
    for i in range(len(y_true)):
        diff_sstot = (y_true[i] - _y) ** 2
        diff_ssres = (y_true[i] - y_predict[i]) ** 2
        ss_tot += diff_sstot
        ss_res += diff_ssres

    score = 1 - (ss_res / ss_tot)

    return score

问题 8 - 交叉验证¶

  • Kk-fold cross-validation
  • GridSearchCV
  • GridSearchCV'cv_results_'

fit_model print pd.DataFrame(grid.cv_results_)

问题 8 - 回答：K折交叉验证是指将整个训练样本集划分为k份，k-1份作为训练集，1份作为验证集，依次轮换训练集和验证集k次，交叉验证重复k次，每个子样本验证一次，平均k次的结果或者使用其它结合方式，最终得到一个单一估测。 cv_results返回一个字典，能告诉我们模型参数，模型性能，模型消耗时间等参数； 如果不适用交叉验证，将无法得到最佳参数组合，并且避免了模型因数据分布不均匀导致的模型训练偏离数据特征的问题，通过交叉验证可以充分避免这种情况出现。 如果有m组参数，会进行k*m次训练和验证。¶

from sklearn.model_selection import KFold,GridSearchCV
from sklearn.tree import DecisionTreeRegressor
from sklearn.metrics import make_scorer

def fit_model(X, y):
    """  [X,y]"""
    
    cross_validator = KFold(n_splits=10)
    
    regressor = DecisionTreeRegressor()

    params = {'max_depth':range(1,11)}

    scoring_fnc = make_scorer(performance_metric)

    grid = GridSearchCV(regressor, params, scoring_fnc, cv=cross_validator)

    #  [X,y]
    grid = grid.fit(X, y)

    # 
    return grid.best_estimator_
